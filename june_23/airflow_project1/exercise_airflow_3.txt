def calculate_average():
    total = 0
    count = 0

    with open("/tmp/marks.txt", "r") as file:
        for line in file:
            mark = int(line.strip().split(",")[1])
            total += mark
            count += 1

    average = total / count

    with open("/tmp/average.txt", "w") as file:
        file.write(str(average))

    print(f"Average = {average}")


def generate_result():
    with open("/tmp/average.txt", "r") as file:
        average = float(file.read())

    result = "PASS" if average >= 50 else "FAIL"

    with open("/tmp/result.txt", "w") as file:

    low_stock = []

    with open("/tmp/inventory.txt", "r") as file:

from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime
import csv


def create_csv():
    with open("/tmp/sales.csv", "w") as file:
        file.write(
"""product,quantity,price
Laptop,2,70000
Mouse,5,500
Keyboard,3,1200"""
        )

    print("sales.csv created")


def read_csv():
    with open("/tmp/sales.csv", "r") as file:
        print(file.read())


def calculate_revenue():
    revenues = {}
    total_revenue = 0

    with open("/tmp/sales.csv", "r") as file:
        reader = csv.DictReader(file)

        for row in reader:
            revenue = int(row["quantity"]) * int(row["price"])

            revenues[row["product"]] = revenue
            total_revenue += revenue

    with open("/tmp/revenue.txt", "w") as file:
        for product, revenue in revenues.items():
            file.write(f"{product}={revenue}\n")

        file.write(f"Total Revenue={total_revenue}")

    print(f"Total Revenue = {total_revenue}")


def create_summary():
    with open("/tmp/revenue.txt", "r") as file:
        content = file.read()

    with open("/tmp/summary.txt", "w") as file:
        file.write(content)

    print("Summary file created")


with DAG(
    dag_id="exercise6_csv_processing",
    start_date=datetime(2025, 1, 1),
    schedule="@daily",
    catchup=False
) as dag:

    create_csv_task = PythonOperator(
        task_id="create_csv",
        python_callable=create_csv
    )

    read_csv_task = PythonOperator(
        task_id="read_csv",
        python_callable=read_csv
    )

    calculate_revenue_task = PythonOperator(
        task_id="calculate_revenue",
        python_callable=calculate_revenue
    )

    create_summary_task = PythonOperator(
        task_id="create_summary",
        python_callable=create_summary
    )

    create_csv_task >> read_csv_task >> calculate_revenue_task >> create_summary_task
