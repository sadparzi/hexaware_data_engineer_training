from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime


def create_department_file():
    with open("/tmp/departments.txt", "w") as file:
        file.write(
"""IT,45000
HR,35000
Finance,50000
IT,55000
Finance,40000
HR,30000"""
        )

    print("Department file created")


def calculate_department_salary():
    department_salary = {}

    with open("/tmp/departments.txt", "r") as file:
        for line in file:
            department, salary = line.strip().split(",")
            salary = int(salary)

            if department in department_salary:
                department_salary[department] += salary
            else:
                department_salary[department] = salary

    with open("/tmp/department_totals.txt", "w") as file:
        for department, total in department_salary.items():
            file.write(f"{department},{total}\n")

    print("Department salary calculated")


def generate_department_report():
    with open("/tmp/department_totals.txt", "r") as file:
        report = file.read()

    with open("/tmp/department_report.txt", "w") as file:
        file.write(report)

    print("Department report generated")


with DAG(
    dag_id="exercise7_department_salary",

from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime
import csv


def create_orders():
    with open("/tmp/orders.csv", "w") as file:
        file.write(
"""product,quantity,price
Laptop,1,70000
Mouse,4,500
Monitor,2,12000
Keyboard,3,1500"""
        )

    print("Orders CSV created")


def calculate_order_value():
    revenues = {}
    total_revenue = 0
    highest_product = ""
    highest_revenue = 0

    with open("/tmp/orders.csv", "r") as file:
        reader = csv.DictReader(file)

        for row in reader:
            revenue = int(row["quantity"]) * int(row["price"])

            revenues[row["product"]] = revenue
            total_revenue += revenue

            if revenue > highest_revenue:
                highest_revenue = revenue
                highest_product = row["product"]

    with open("/tmp/order_values.txt", "w") as file:
        for product, revenue in revenues.items():
            file.write(f"{product} = {revenue}\n")

        file.write(f"Total Revenue = {total_revenue}\n")
        file.write(f"Highest Selling Product = {highest_product}")


def generate_sales_report():
    with open("/tmp/order_values.txt", "r") as file:
        report = file.read()

    with open("/tmp/sales_report.txt", "w") as file:
        file.write(report)

    print("Sales report generated")

def calculate_balance():
    total_deposit = 0
    total_withdrawal = 0

    with open("/tmp/transactions.txt", "r") as file:
